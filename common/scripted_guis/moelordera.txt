scripted_gui = {
	moelord_welcome = {
		context_type = player_context
		window_name = "moelord_welcome_container"

		visible = {
			has_country_flag = moelord_welcome
		}

		effects = {
			welcome_close_button_click = {
				clr_country_flag = moelord_welcome
				country_event = update_status.1
			}
		}
	}

	moelord_help_button = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "moelord_help_button_container"

		visible = {
			OR = {
				tag = CSY
				tag = CYX
				tag = JAP
			}
		}

		effects = {
			moelord_help_button_click = {
				if = {
					limit = { NOT = { has_variable = moelord_help_tab_on } }
					set_variable = { moelord_help_tab_on = 1 }
				}
				else = {
					clear_variable = moelord_help_tab_on
				}
			}
		}
	}

	moelord_help_tab = {
		context_type = player_context
		parent_window_name = "time_control_window"
		window_name = "moelord_help_tab_container"

		visible = {
			is_ai = no
			has_variable = moelord_help_tab_on
		}

		effects = {
			moelord_button_a_click = {
				CYX = {
					set_country_leader_name = { name = "Hoshimachi Suisei" }
					every_unit_leader = {
						if = {
							limit = { has_trait = General_is_a_HoS }
							set_leader_name = "Hoshimachi Suisei"
						}
					}
				}
			}
			moelord_button_b_click = {
				CSY = {
					set_country_leader_name = { name = "Hakurei Reimu" }
					every_unit_leader = {
						if = {
							limit = { has_trait = General_is_a_HoS }
							set_leader_name = "Hakurei Reimu"
						}
					}
				}
			}
			moelord_button_c_click = {
				JAP = {
					set_country_leader_name = { name = "Hoshimachi Suisei" }
					set_country_leader_portrait = { portrait = "gfx/leaders/China/Portrait_China_Liu_Zhenhua.tga" }
					change_tag_from = CYX
				}
				CYX = {
					every_unit_leader = {
						if = {
							limit = { has_trait = General_is_a_HoS }
							set_nationality = JAP
						}
					}
				}
			}
			moelord_button_d_click = {
				JAP = {
					set_country_leader_name = { name = "Hakurei Reimu" }
					set_country_leader_portrait = { portrait = "gfx/leaders/China/Portrait_China_Cai_Chengxun.tga" }
					change_tag_from = CSY
				}
				CSY = {
					every_unit_leader = {
						if = {
							limit = { has_trait = General_is_a_HoS }
							set_nationality = JAP
						}
					}
				}
			}
			moelord_button_e_click = {
				CSC = { play_song = "little_busters" }
			}
			moelord_button_f_click = {
				every_country = {
					limit = { is_ai = no }
					play_song = "little_busters"
				}
			}
			moelord_button_g_click = {
				every_country = {
					limit = { is_ai = no }
					play_song = "little_busters"
				}
			}
			moelord_button_h_click = {
				every_country = {
					set_country_leader_name = { name = "Hakurei Reimu" }
					set_country_leader_portrait = { portrait = "gfx/leaders/China/Portrait_China_Cai_Chengxun.tga" }
					every_unit_leader = {
						set_leader_name = "Hakurei Reimu"
						set_leader_portrait = "gfx/leaders/China/Portrait_China_Cai_Chengxun.tga"
					}
					play_song = "maintheme"
				}
			}
		}
	}
}
